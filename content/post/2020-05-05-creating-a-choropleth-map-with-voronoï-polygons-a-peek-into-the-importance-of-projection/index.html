In this article I want two things : how to build a choropleth map with Vornoï polygon using sf. The map I will be drawing is the area in which each Ligue 1 football team is the nearest in France. First, I will use the recommanded legal projection for France, which is Lambert projection (crs : 2154). Then I will show the results using the WGS64 projection which is used to share geospatial data.

get_location <- function(city) {
  city %>% 
    paste0("https://api-adresse.data.gouv.fr/search/?q=",.) %>%
    GET() %>%
    pluck("content") %>%
    rawToChar() %>%
    fromJSON() %>%
    pluck("features") %>% 
    flatten() %>%
    slice(1) %>% 
    pluck("geometry.coordinates", 1) %>% 
    set_names(c("LON", "LAT")) %>% 
    list()
}


clubs_geo <- clubs %>% 
  rowwise() %>% 
  mutate(coords = get_location({city})) %>% 
  unnest_wider(coords) %>% 
  # I manually add Monaco which is not in France
  bind_rows(tibble(city = "Monaco", LON = 7.414793, LAT = 43.728353)) %>% 
  st_as_sf(coords = c("LON", "LAT")) %>% 
  st_set_crs(4326) %>% 
  st_transform(2154)
  
clubs_geo <- clubs_geo %>% 
  bind_cols(as_tibble(
    st_coordinates(clubs_geo)
  ))



#France
france <- st_read(
  here("static/inputs/ligue 1 voronoi/metropole-version-simplifiee.geojson")
  ) %>%
  st_transform(2154) %>%
  st_cast("POLYGON") %>%
  mutate(area = st_area(geometry)) %>%
  filter(as.numeric(area) == max(as.numeric(area)))

voronoi <- clubs_geo %>%
  st_union() %>%
  st_voronoi() %>%
  st_cast() %>%
  st_sf() %>% 
  st_join(clubs_geo, join = st_contains) %>% 
  st_intersection(france) %>% 
  mutate(area = st_area(geometry))

library(raster)


r <- raster(france, ncol = 500, nrow = 500)
distances <- distanceFromPoints(r, clubs_geo) %>% 
  mask(france) %>% 
  as.data.frame(xy = TRUE) %>% 
  na.omit() %>% 
  as_tibble() %>% 
  mutate(distance = layer / 1000) %>% 
  dplyr::select(-layer) %>% 
  mutate(bins = case_when(
           distance < 50 ~ "<50 km",
           distance <= 100 ~ "50 à 100 km",
           distance <= 150 ~ "100 à 150 km",
           distance <= 200 ~ "150 à 200 km",
           distance >200 ~ ">200 km"),
         bins = fct_reorder(bins, distance)) 
 
description_2 <- "Distance in kms to the nearest Ligue 1 football club"

ggplot() +
  geom_raster(data = distances, aes(fill = bins,
                                    x = x,
                                    y = y),
              interpolate = TRUE) +
  scale_fill_brewer(name = NULL) +
  geom_sf(data = clubs_geo) +
  geom_sf(data = france,
          fill = "transparent") +
  geom_label_repel(data = clubs_geo,
                   aes(x = X,
                       y = Y,
                       label = city)) +
  annotate("text", x = -Inf, y = Inf, 
           label = description_2,
